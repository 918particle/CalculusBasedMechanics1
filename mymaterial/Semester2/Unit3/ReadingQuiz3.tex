\section{Memory Bank}

\begin{itemize}
\item $V = IR$ ... Ohm's Law
\item $R^{-1}_{\rm tot} = R_1^{-1} + R_2^{-1} + ...$ ... Adding resistors in parallel.
\item $R_{\rm tot} = R_1 + R_2 + ...$ ... Adding resistors in series.
\item $P = IV$ ... Power in DC circuits
\end{itemize}
